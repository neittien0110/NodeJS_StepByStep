/**
 * Loại bỏ dấu khỏi chuỗi văn bản tiếng Việt.
 * @customfunction VanBanKhongDau
 * @param {string} text Văn bản tiếng Việt có dấu. Ví dụ Lê Văn Long
 * @returns {string} văn bản không dấu
 * @helpurl https://users.soict.hust.edu.vn/sinno/projects/myexcel/help/global.html#VanBanKhongDau
 */
function VanBanKhongDau(text) {
    //Danh sách các kí tự tiếng Việt
    var VietnameseSigns = [
        "aAeEoOuUiIdDyY",
        "áàạảãâấầậẩẫăắằặẳẵ",
        "ÁÀẠẢÃÂẤẦẬẨẪĂẮẰẶẲẴ",
        "éèẹẻẽêếềệểễ",
        "ÉÈẸẺẼÊẾỀỆỂỄ",
        "óòọỏõôốồộổỗơớờợởỡ",
        "ÓÒỌỎÕÔỐỒỘỔỖƠỚỜỢỞỠ",
        "úùụủũưứừựửữ",
        "ÚÙỤỦŨƯỨỪỰỬỮ",
        "íìịỉĩ",
        "ÍÌỊỈĨ",
        "đ",
        "Đ",
        "ýỳỵỷỹ",
        "ÝỲỴỶỸ"
    ];
    var regex;
    for (var i = 1; i < VietnameseSigns.length; i++) {
        for (var j = 0; j < VietnameseSigns[i].length; j++) {
            // Fix-bug: không đổi được toàn bộ kí tự.
            // Nguyên nhân: Hàm replace với tham số vào string chỉ cho phép đổi 1,2 trường hợp đầu tiên
            //              Muốn thay thế toàn bộ, phải sử dụng tham số dạng Regex
            regex = new RegExp(VietnameseSigns[i][j], "g");
            text = text.replace(regex, VietnameseSigns[0][i - 1]);
        }
    }
    return text;
}
exports.VanBanKhongDau = VanBanKhongDau;
/**
 * Lấy duy nhất từ cuối cùng khỏi tên đầy đủ =MY.LAYTEN("Lê Văn Long")
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Từ cuối cùng
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractLastName(FullName) {
    var i;
    var len;
    try {
        /// Loại bỏ kí tự trống ở cuối nếu có
        FullName = String(FullName).trimRight();
        len = FullName.length;
        /// Tìm vị trí kí tự space cuối cùng
        for (i = len - 1; i >= 0; i--) {
            if (FullName[i] == " ")
                break;
        }
        if (i < 0) {
            return "";
        }
        else {
            return FullName.substr(i + 1, len - i - 1);
        }
    }
    catch (err) {
        return err.message;
    }
}
/**
 * Lấy duy nhất từ đầu tiên khỏi tên đầy đủ
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Từ đầu tiên
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractFirstName(FullName) {
    var i;
    var len;
    /// Loại bỏ kí tự trống ở đầu nếu có
    FullName = FullName.trimLeft();
    len = FullName.length;
    /// Tìm vị trí kí tự space đầu tiên
    for (i = 1; i < len; i++) {
        if (FullName[i] == " ")
            break;
    }
    if (i < len) {
        return FullName.substr(0, i);
    }
    else {
        return FullName;
    }
}
/**
 * Lấy duy nhất đệm  khỏi tên đầy đủ
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Đệm
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractMiddleName(FullName) {
    var pos_end;
    var pos_start;
    var len;
    /// Loại bỏ kí tự trống ở cả 2 đầu cuối nếu có
    FullName = FullName.trim();
    len = FullName.length;
    /// Tìm vị trí kí tự space đầu tiên
    for (pos_start = 1; pos_start < len; pos_start++) {
        if (FullName[pos_start] == " ")
            break;
    }
    if (pos_start >= len) {
        return "";
    }
    /// Bỏ qua các kí tự trống
    for (pos_start++; pos_start < len && FullName[pos_start] == " "; pos_start++) { }
    /// Tìm vị trí kí tự space cuối cùng
    for (pos_end = len - 1; pos_end > pos_start; pos_end--) {
        if (FullName[pos_end] == " ")
            break;
    }
    if (pos_end < pos_start) {
        return "";
    }
    else {
        return FullName.substring(pos_start, pos_end).trimRight();
    }
}
/**
 * Lấy các chữ cái đầu tên
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Chuỗi các chữ cái đầu
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractFirstLetters(FullName) {
    /// Kết quả cần tính
    var ShorthenName = "";
    /// Loại bỏ kí tự trống ở cả 2 đầu cuối nếu có
    FullName = FullName.trim();
    var len = FullName.length;
    if (len == 1) {
        /// Nếu chỉ có 1 kí tự thì xong luôn
        ShorthenName = FullName;
    }
    else {
        /// Phân tách từ và chữ cái đầu của từng từ
        ShorthenName = ShorthenName + FullName[0];
        for (var i = 0; i < len; i++) {
            if (FullName[i] == " " && i < len - 1 && FullName[i + 1] != " ") {
                ShorthenName = ShorthenName + FullName[i + 1];
                i++;
            }
        }
    }
    return ShorthenName;
}
