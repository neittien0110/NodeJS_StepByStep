/**
 * Email HUST do nhà trường cấp cho mỗi sinh viên
 * @customfunction HUST.EmailSinhVien
 * @param {string} HoVaTen Họ và tên đầy đủ. Ví dụ Lê Văn Long.
 * @param {string} MaSoSinhVien Mã số SV do trường cấp. Bỏ trống nếu là giảng viên. Ví dụ 20202987.
 * @returns {string} Email HUST của sinh viên. Ví dụ long.lv202987@sis.hust.edu.vn
 * @example =StudentEmail("Lê Văn Long", 20202987) =long.lv202987@sis.hus.edu.vn
 * @helpurl https://mycel.app/help/StudentEmail.html
 * @interface
*/
function StudentEmail(HoVaTen, MaSoSinhVien) {
    var TenKhongDau = LayTenKhongDau(HoVaTen);
    var Email;
    var MSSV;
    MSSV = String(MaSoSinhVien);
    if (MSSV.length == 8) {
        ///Công thức dành cho email sinh viên
        var ChuCaiDau = LayCacChuCaiDau(TenKhongDau);
        Email = LayTen(TenKhongDau) + '.' + ChuCaiDau.substring(0, ChuCaiDau.length - 1) + MSSV.substring(2, MSSV.length) + "@sis.hust.edu.vn";
    }
    else if (MSSV == "undefined" || MSSV.length == 0) {
        ///Công thức dành cho email giáo viên
        Email = LayTen(TenKhongDau) + "." + LayHo(TenKhongDau) + LayDem(TenKhongDau).trim() + "@hust.edu.vn";
    }
    else {
        Email = "Mã số không hợp lệ";
    }
    return Email;
}
/**
 * Thời gian bắt đầu của mỗi kíp thi
 * @customfunction HUST.KipThi
 * @param {number} KipThi   Số thứ tự kíp thi, là 1 | 2 | 3 | 4. Bách Khoa Hà Nội chỉ có 4 buổi thi mỗi ngày.
 * @param {string} PhanTach Kí tự phân tách giữa giờ và phút. Mặc định là ':'
 * @returns {string} Giờ bắt đầu của kíp thi
 * @example =KipThi(2) =7:00
 *          =KipThi(3," giờ "") =12 giờ 30
 * @helpurl https://mycel.app/help/KipThi.html
 * @interface
*/
function KipThi(KipThi, PhanTach) {
    if (PhanTach === void 0) { PhanTach = ":"; }
    if (PhanTach === null) {
        PhanTach = ":";
    }
    var StartTime;
    switch (KipThi) {
        case 1:
            StartTime = "7" + PhanTach + "00";
            break;
        case 2:
            StartTime = "9" + PhanTach + "30";
            break;
        case 3:
            StartTime = "12" + PhanTach + "30";
            break;
        case 4:
            StartTime = "15" + PhanTach + "00";
            break;
        default:
            StartTime = "Không hợp lệ";
            break;
    }
    return StartTime;
}

/**
 * Qui đổi điểm số ở thang đo 10 điểm về thang đo 4
 * @customfunction HUST.ThangDiem10_4
 * @param {number} d10 Điểm số ở thang đo 10. Ví dụ 8.0
 * @returns {number} Điểm số ở thang đo 4. Ví dụ 3.5
 * @example =ThangDiem10_4(9.5) =4.0
 * @helpurl https://mycel.app/help/ThangDiem10_4.html
 * @interface
*/
function ThangDiem10_4(d10) {
    var x; // Điểm số ở thang ABCD
    if (d10 >= 8.5 && d10 <= 10)
        x = 4.0;
    else if (d10 >= 8 && d10 < 8.5)
        x = 3.5;
    else if (d10 >= 7 && d10 < 8)
        x = 3.0;
    else if (d10 >= 6.5 && d10 < 7)
        x = 2.5;
    else if (d10 >= 5.5 && d10 < 6.5)
        x = 2.0;
    else if (d10 >= 5 && d10 < 5.5)
        x = 1.5;
    else if (d10 >= 4 && d10 < 5)
        x = 1.0;
    else if (d10 < 4)
        x = 0.0;
    return x;
}
/**
 * Qui đổi điểm số ở thang đo 10 điểm về thang đo ABCD
 * @customfunction HUST.ThangDiem10_ABCD
 * @param {number} d10 Điểm số ở thang đo 10. Ví dụ 8.5
 * @returns {string} Điểm số ở thang đo 4. Ví dụ A
 * @example =ThangDiem10_ABCD(9.5) =A+
 * @helpurl https://mycel.app/help/ThangDiem10_ABCD.html
 * @interface
*/
function ThangDiem10_ABCD(d10) {
    var x; // Điểm số ở thang ABCD
    if (d10 >= 9.5 && d10 <= 10)
        x = 'A+';
    else if (d10 >= 8.5 && d10 < 9.5)
        x = 'A';
    else if (d10 >= 8 && d10 < 8.5)
        x = 'B+';
    else if (d10 >= 7 && d10 < 8)
        x = 'B';
    else if (d10 >= 6.5 && d10 < 7)
        x = 'C+';
    else if (d10 >= 5.5 && d10 < 6.5)
        x = 'C';
    else if (d10 >= 5 && d10 < 5.5)
        x = 'D+';
    else if (d10 >= 4 && d10 < 5)
        x = 'D';
    else if (d10 < 4)
        x = 'F';
    return x;
}
/**
 * Thời gian bắt đầu/kết thúc của các tiết học
 * @customfunction HUST.TietHoc
 * @param {number} Period Tiết học bắt đầu. Phạm vi [1, 14]. Ví dụ 1; 3;
 * @param {boolean} Start Xem giờ bắt đầu (true) hoặc giờ kết thúc (false). Mặc định true.
 * @param {string} [Seperator] Kí tự phân tách giữa giờ và phút. Mặc định là ':'
 * @returns {string} Thời gian bắt đầu hoặc kết thúc của 1 tiết học hoặc nhiều tiết học.
 * @example =TietHoc(2) =7h00
 * @helpurl https://mycel.app/help/TietHoc.html
 * @interface
*/
function TietHoc(Period, Start, Seperator) {
    if (Start === void 0) { Start = true; }
    if (Seperator === void 0) { Seperator = ':'; }
    var TimeTable = [['6:45', '7:30'],
        ['7:30', '8:15'],
        ['8:25', '9:10'],
        ['9:20', '10:05'],
        ['10:15', '11:00'],
        ['11:00', '11:45'],
        ['12:30', '13:15'],
        ['13:15', '14:00'],
        ['14:10', '14:55'],
        ['15:05', '15:50'],
        ['16:00', '16:45'],
        ['16:45', '17:30'],
        ['17:45', '18:30'],
        ['18:30', '19:15']
    ];
    if (Period < 1 || Period > 14) {
        var error = new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidValue, "Tiết học chỉ hợp lệ từ 1 tới 14");
        throw error;
    }
    if (Seperator === null) {
        Seperator = ":";
    }
    if (Start === null) {
        Start = true;
    }
    // Trích ra thời gian bắt đầu/kết thúc từ bảng hằng Thời khóa biểu  
    return TimeTable[Period][Start ? 0 : 1].replace(":", Seperator);
}
/**
 * Chuyển đổi từ email thành mã số sinh viên
 * @customfunction HUST.Email2MaSo
 * @param {string} Email HUST của sinh viên. Ví dụ dat.hdt20121987@sis.hust.edu.vn
 * @returns {number} MaSoSinhVien Mã số SV do trường cấp. Bỏ trống nếu là giảng viên. Ví dụ 20202987.
 * @example =Email2MaSo("dat.hdt121987@sis.hust.edu.vn") = 20121987
 * @helpurl https://mycel.app/help/Email2MaSo.html
 * @throws {invalidValue} gia trì đầu vào không phải là email sinh viên
 * @interface
 */
function Email2MaSo(email) {
    // Xác định vị trí của kí tự @
    var PosOfAt;
    PosOfAt = email.indexOf("@");
    if (PosOfAt < 0) {
        throw new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidValue, "Không phải email hợp lệ");
    }
    // Nếu không đủ độ dài 6 số thì cũng không hợp lệ
    if (PosOfAt < 6) {
        throw new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidValue, "Thông tin trích xuất mã số sinh viên không phù hợp");
    }
    // trích xuất 6 kí tự tiềm năng là MSSV
    var suspects;
    suspects = email.substring(PosOfAt - 6, PosOfAt);
    var result;
    result = Number(suspects);
    if (result === NaN) {
        throw new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidNumber, "Mã số sinh viên phải có dạng số nguyên 8 chữ số");
    }
    return result + 20000000;
}
module.exports.Email2MaSo = Email2MaSo;
