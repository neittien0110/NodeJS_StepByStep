/**
 * Lấy duy nhất từ cuối cùng khỏi tên đầy đủ =MY.LAYTEN("Lê Văn Long")
 * @private

 * @returns {string} Từ cuối cùng
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractLastName(FullName) {
    var i;
    var len;
    try {
        /// Loại bỏ kí tự trống ở cuối nếu có
        FullName = String(FullName).trimRight();
        len = FullName.length;
        /// Tìm vị trí kí tự space cuối cùng
        for (i = len - 1; i >= 0; i--) {
            if (FullName[i] == " ")
                break;
        }
        if (i < 0) {
            return "";
        }
        else {
            return FullName.substr(i + 1, len - i - 1);
        }
    }
    catch (err) {
        return err.message;
    }
}

/**
 * Lấy duy nhất đệm  khỏi tên đầy đủ
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Đệm
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractMiddleName(FullName) {
    var pos_end;
    var pos_start;
    var len;
    /// Loại bỏ kí tự trống ở cả 2 đầu cuối nếu có
    FullName = FullName.trim();
    len = FullName.length;
    /// Tìm vị trí kí tự space đầu tiên
    for (pos_start = 1; pos_start < len; pos_start++) {
        if (FullName[pos_start] == " ")
            break;
    }
    if (pos_start >= len) {
        return "";
    }
    /// Bỏ qua các kí tự trống
    for (pos_start++; pos_start < len && FullName[pos_start] == " "; pos_start++) { }
    /// Tìm vị trí kí tự space cuối cùng
    for (pos_end = len - 1; pos_end > pos_start; pos_end--) {
        if (FullName[pos_end] == " ")
            break;
    }
    if (pos_end < pos_start) {
        return "";
    }
    else {
        return FullName.substring(pos_start, pos_end).trimRight();
    }
}
/**
 * Lấy các chữ cái đầu tên
 * @private
 * @param {string} FullName Tên đầy đủ
 * @returns {string} Chuỗi các chữ cái đầu
 * @helpurl http://soict.hust.edu.vn/sinno/
 */
function ExtractFirstLetters(FullName) {
    /// Kết quả cần tính
    var ShorthenName = "";
    /// Loại bỏ kí tự trống ở cả 2 đầu cuối nếu có
    FullName = FullName.trim();
    var len = FullName.length;
    if (len == 1) {
        /// Nếu chỉ có 1 kí tự thì xong luôn
        ShorthenName = FullName;
    }
    else {
        /// Phân tách từ và chữ cái đầu của từng từ
        ShorthenName = ShorthenName + FullName[0];
        for (var i = 0; i < len; i++) {
            if (FullName[i] == " " && i < len - 1 && FullName[i + 1] != " ") {
                ShorthenName = ShorthenName + FullName[i + 1];
                i++;
            }
        }
    }
    return ShorthenName;
}

                MSSV.substring(2, MSSV.length) +
                "@sis.hust.edu.vn";
    }

    else {
        Email = "Mã số không hợp lệ";

 * @helpurl https://mycel.app/help/TietHoc.html
 * @interface
 */
function TietHoc(Period, Start, Seperator) {
    if (Start === void 0) { Start = true; }

        ["6:45", "7:30"],

        ["15:05", "15:50"],
        ["16:00", "16:45"],
        ["16:45", "17:30"],
        ["17:45", "18:30"],
        ["18:30", "19:15"]
    ];
    if (Period < 1 || Period > 14) {
        var error = new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidValue, "Tiết học chỉ hợp lệ từ 1 tới 14");
        throw error;
    }
    if (Seperator === null) {
        Seperator = ":";
    }
    if (Start === null) {
        Start = true;
    }
    // Trích ra thời gian bắt đầu/kết thúc từ bảng hằng Thời khóa biểu

function Dec2Bin2(Number, BitNum, Sign) {
    var DoDaiBit;
    DoDaiBit = BitNum === null ? 8 : BitNum;
    var CoDau;
    CoDau = Sign === null ? true : Sign;
    console.log("Number = %d; DoDaiBit = %i, CoDau = %s", Number, DoDaiBit, CoDau);
    if (DoDaiBit >= 32) {
        var error = new CustomFunctions.Error(CustomFunctions.ErrorCode.invalidValue, "Số bit phải < 32");
        throw error;
    }
    /// Kiểm tra điều kiện phạm vi biểu diễn
    var UpperBoundary;
    var LowerBoundary;
    if (CoDau) {
        UpperBoundary = 1 << (DoDaiBit - 1);
        LowerBoundary = -UpperBoundary;
        UpperBoundary = UpperBoundary - 1;
    }
    else {
        UpperBoundary = (1 << DoDaiBit) - 1;
        LowerBoundary = 0;
    }
    if (Number < LowerBoundary || Number > UpperBoundary) {
        return "Out of range [" + LowerBoundary + "," + UpperBoundary + "]";
    }
    /// Chuyển ra hệ nhị phân
    var res;
    if (Number >= 0) {
        res = Number.toString(2).padStart(DoDaiBit, "0");
    }
    else {
        // Tìm ra số tương ứng
        var InvNum = (1 << DoDaiBit) + Number;
        // Bù 2
        res = InvNum.toString(2).padStart(DoDaiBit, "0");
    }
    return res;
}

            for (index++; index < list.length; index++) {

        });

    }
